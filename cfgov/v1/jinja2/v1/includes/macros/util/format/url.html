{# ========================================================================== slugify() ========================================================================== Description: Formats a string to: 'foo-bar' When given: string: A string, such as 'Foo Bar'. ========================================================================== #} {% macro slugify(string) -%} {{ string | lower | replace(' ', '-') | replace(',', '') | replace('.', '') | replace('!', '') | replace('?', '') }} {%- endmacro %}
